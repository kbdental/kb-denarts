GOOGLE SHEETS CONNECTION
This app connects to Google Sheets via a Google Apps Script Web App. Orders, doctors, and billing data sync automatically when you save or update.
Paste the Web App URL after deploying the script below
WHAT GETS SYNCED
✓ Orders Sheet
Model No, Challan, Date, Doctor, Patient, Work Type, Teeth, Units, Due Date, Status, Amount, Billing Status
✓ Doctors Sheet
Doctor Name, Clinic, Address, Phone, Email, Contact Person
⬆ Push (App → Sheet): Saves every time you create/update an order or master.
⬇ Pull (Sheet → App): Click "Pull from Sheet" to import changes made directly in Google Sheets.
Orders newer in the sheet override app data. New sheet rows are added to the app.
SETUP INSTRUCTIONS
Step 1. Open your Google Sheet → go to Extensions → Apps Script
Step 2. Delete any existing code, paste the script below, click Save
Step 3. Click Deploy → New Deployment → Web App
Step 4. Set Execute as: Me  |  Who has access: Anyone → Deploy
Step 5. Copy the Web App URL, paste it above → Save & Test
Step 6. Click Full Sync to push all 110 existing orders to your sheet
// ═══════════════════════════════════════════ // K.B. DENARTS — Google Sheets Sync Script // Paste this in Extensions → Apps Script // ═══════════════════════════════════════════ const ORDERS_SHEET = 'Orders'; const DOCTORS_SHEET = 'Doctors'; function doPost(e) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var data = JSON.parse(e.postData.contents); if (data.type === 'order') { var sheet = getOrCreateSheet(ss, ORDERS_SHEET, ['Model No','Challan No','Date','Received Date','Due Date','Dispatch Date', 'Doctor','Clinic','Patient','Work Type','Teeth','Units','Status', 'Amount (₹)','Billing Status','Implant System','Notes']); // Find existing row by Model No to update, else append var modelNo = data.modelNo; var lastRow = sheet.getLastRow(); var found = false; if (lastRow > 1) { var modelCol = sheet.getRange(2, 1, lastRow - 1, 1).getValues(); for (var i = 0; i < modelCol.length; i++) { if (modelCol[i][0] == modelNo) { sheet.getRange(i + 2, 1, 1, 17).setValues([[ data.modelNo, data.challanNo||'', data.date||'', data.receivedDate||'', data.dueDate||'', data.dispatchDate||'', data.doctorName||'', data.clinicName||'', data.patientName||'', data.workType||'', data.teeth||'', data.units||0, data.status||'', data.amount||0, data.billingStatus||'', data.implantSystem||'', data.notes||'' ]]); found = true; break; } } } if (!found) { sheet.appendRow([ data.modelNo, data.challanNo||'', data.date||'', data.receivedDate||'', data.dueDate||'', data.dispatchDate||'', data.doctorName||'', data.clinicName||'', data.patientName||'', data.workType||'', data.teeth||'', data.units||0, data.status||'', data.amount||0, data.billingStatus||'', data.implantSystem||'', data.notes||'' ]); } } if (data.type === 'doctor') { var dsheet = getOrCreateSheet(ss, DOCTORS_SHEET, ['Doctor Name','Clinic','Address','Phone','Email','Contact Person','CP Phone','Category']); var name = data.name; var lastRow2 = dsheet.getLastRow(); var found2 = false; if (lastRow2 > 1) { var nameCol = dsheet.getRange(2, 1, lastRow2 - 1, 1).getValues(); for (var j = 0; j < nameCol.length; j++) { if (nameCol[j][0] == name) { dsheet.getRange(j + 2, 1, 1, 8).setValues([[ data.name, data.clinic||'', data.address||'', data.phone||'', data.email||'', data.cpName||'', data.cpPhone||'', data.category||'' ]]); found2 = true; break; } } } if (!found2) { dsheet.appendRow([data.name, data.clinic||'', data.address||'', data.phone||'', data.email||'', data.cpName||'', data.cpPhone||'', data.category||'']); } } return ContentService .createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({status:'error',message:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService .createTextOutput(JSON.stringify({status:'connected',app:'KB DENARTS'})) .setMimeType(ContentService.MimeType.JSON); } function getOrCreateSheet(ss, name, headers) { var sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); sheet.appendRow(headers); sheet.getRange(1, 1, 1, headers.length) .setBackground('#1d4ed8').setFontColor('#ffffff').setFontWeight('bold'); } return sheet; }
